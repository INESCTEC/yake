# 1. Lematização com agregação MIN (melhor score)
wsl ~/.local/bin/yake -ti 'Trees are tall. Many trees grow. The tree provides shade.' --lemmatize --lemma-aggregation min -t 5 -v

# 2. Lematização com agregação MEAN (média)
wsl ~/.local/bin/yake -ti 'running runs ran runner' --lemmatize --lemma-aggregation mean -t 3 -v

# 3. Lematização com agregação MAX (pior score)
wsl ~/.local/bin/yake -ti 'studies studied studying research' --lemmatize --lemma-aggregation max -t 3 -v

# 4. Lematização com agregação HARMONIC (balanceada)
wsl ~/.local/bin/yake -ti 'The studies studied studying researchers research' --lemmatize --lemma-aggregation harmonic -t 3 -v

# 5. Comparar COM e SEM lematização
wsl ~/.local/bin/yake -ti 'Trees grow. Many trees. The tree is tall.' -t 5 -v
wsl ~/.local/bin/yake -ti 'Trees grow. Many trees. The tree is tall.' --lemmatize -t 5 -v
